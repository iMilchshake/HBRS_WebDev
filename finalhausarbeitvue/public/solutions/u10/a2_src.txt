<?php
if (isset($_POST['user']) && isset($_POST['pass'])) {
    $login = $_POST['user'] . ',' . $_POST['pass'] . "\n";
    $file = './raw_passwords.csv';
    $lines = file($file);

    $success = false;
    foreach($lines as $index => $line) {
        if($line == $login) {
            $success = true;
            break;
        }
    }

    if($success) {
        echo 'success!';
    } else {
        echo 'invalid login!';
    }
}
